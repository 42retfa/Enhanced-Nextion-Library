#include "NexCrop.h"
NexCrop q0 = NexCrop(0, 1, "q0");
NexTouch *nex_listen_list[] =
{
&q0,
NULL
};
void q0PopCallback(void *ptr)
{
uint32_t number = 0;
dbSerialPrintln("q0PopCallback");
q0.getPic(&number);
number += 1;
number %= 2;
q0.setPic(number);
}
void setup(void)
{
nexInit();
q0.attachPop(q0PopCallback);
dbSerialPrintln("setup done");
}
void loop(void)
{
nexLoop(nex_listen_list);
}
